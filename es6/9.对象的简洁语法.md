    //传统对象_单体模式写法 key-value模式
    var person = {
        name:'jam',
        age:28,
        showName:function(){
            return this.name;
        },
        showAge:function(){
            return this.age;
        }
    };
    // 调用
    console.log(person.showName()); // jam
    console.log(person.showAge()); // 28

        },

        }
    };
